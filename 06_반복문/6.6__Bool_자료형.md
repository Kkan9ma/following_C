## 관계 연산자 사용 시 주의할 점 1

            int i; // 선언
            i = 5; // 대입
            i == 5; // 비교 (while, if등 괄호에 넣을 수 있음.)
            5 = i; // literal constant에 변수를 넣으려 하는 것이기에 오류.
            5 == i; // 비교니 오류 아님.

## 주의할 점 2

            while (i = 5){}; // 웬만하면 실수
            while (i == 5){}; // 올바른 예
