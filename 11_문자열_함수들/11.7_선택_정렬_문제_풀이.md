            void selectionSort(int arr[], int n)
            {
                int i, j, min_idx;

                min_idx = 0;

                for (i = 0; i < n - 1; i++)
                {
                    for (j = i + 1; j < n; j++)
                    {
                        if (arr[i] > arr[j]) {
                            min_idx = j;
                            swap(arr + i, arr + min_idx);
                        }
                    }
                }
            }

            void selectionSort(int arr[], int n)

                    min_idx = i;
                    for (j = i + 1; j < n; j++)
