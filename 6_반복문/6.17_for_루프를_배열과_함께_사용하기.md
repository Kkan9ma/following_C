            #include <stdio.h>

            #define SIZE 5

            int main()
            {
                int arr[SIZE];
                int i = 0;
                int sum = 0;

                printf("Enter %d numbers : ", SIZE);
                
                for (i = 0; i < SIZE; i++) {
                    scanf("%d", &arr[i]);
                }

                for (i = 0; i < SIZE; i++) {
                    sum += arr[i];
                }

                printf("Sum = %d\n", sum);
                
                return 0;
            }
